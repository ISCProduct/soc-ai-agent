# --- LangChain エコシステム ---
# crewai は依存衝突のため requirements.txt から除外済み（langchain 上限は crewai 制約によらない）。
# 2026-03 に 0.3.x 系列へ移行（#135）。0.4.0 以降は破壊的変更のリスクがあるため上限を設定。
langchain>=0.2.16,<0.4.0
langchain-community>=0.4.2,<0.5.0
langchain-core>=0.3.86,<0.4.0
# langchain-openai 0.3.x が langchain 0.3.x + openai 2.x に対応するバージョン系列
langchain-openai>=0.3.0,<0.4.0
langchain-text-splitters>=0.3.11,<0.4.0

# --- ベクトルストア ---
# main.py で PersistentClient / get_or_create_collection / upsert / query を直接利用。
# これらの API は 0.6.x で互換性を確認済み。
# 0.7.0 以降は API 変更リスクがあるため、requirements.txt と合わせ <0.7.0 に固定。
chromadb>=0.6.3,<0.7.0

# --- 数値計算 ---
# numpy 2.0 以降で chromadb および一部パッケージとの互換性問題が報告されているため上限を維持。
numpy<3.0

# --- LLM プロキシ ---
# litellm は langchain-community の推移的依存。メジャーバージョン固定で API 互換性を保証。
litellm>=1.83.7,<2.0

# --- OpenAI SDK ---
# litellm>=1.83.7 が openai 2.x を要求するため 2.x 系列に移行。
# client.responses / chat.completions / embeddings API は 2.x でも互換性あり。
openai>=1.0,<3.0

# --- トークナイザー ---
# tiktoken: OpenAI エンコーダーとのバイト互換を保つため 0.13.x に固定。
# 0.14.x 以降はエンコーディング変更の可能性あり、アップグレード時は動作確認が必要。
tiktoken>=0.13.0,<0.14

# --- HTTP クライアント ---
# openai SDK との互換性が確認された 0.28.x 系列に固定。
# 0.29.x 以降は再検証が必要なため上限を設定。
httpx>=0.28.1,<0.29
